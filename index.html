// Fungsi untuk download dengan pilihan resolusi // Karena kita hanya memiliki satu sumber video (misal vidio 1.mp4), // kita akan tetap menggunakan file yang sama namun menambahkan label resolusi pada nama file. // Untuk simulasi lebih baik, kita bisa memetakan resolusi ke file berbeda jika ada, // tapi dalam kasus ini kita unduh file original dengan penamaan sesuai kualitas. function downloadWithQuality(quality) { // Pilih video sumber (bisa menggunakan video utama pertama sebagai contoh) // Atau bisa juga memilih video default yaitu "vidio 1.mp4" let baseVideoUrl = "vidio 1.mp4"; // default // Cek apakah ada video yang sedang diputar atau video pertama di galeri const anyVideo = document.querySelector('.video-card video'); if (anyVideo && anyVideo.src) { baseVideoUrl = anyVideo.src; } // Buat nama file berdasarkan resolusi let resolutionLabel = ""; if (quality === "360") resolutionLabel = "360p"; else if (quality === "720") resolutionLabel = "720p"; else if (quality === "1080") resolutionLabel = "1080p"; const fileName = `video_${resolutionLabel}_${Date.now()}.mp4`; // Tampilkan notifikasi ringan (opsional) const btn = document.getElementById('downloadQualityBtn'); const originalText = btn.innerHTML; btn.innerHTML = '⏳ Mengunduh...'; setTimeout(() => { btn.innerHTML = originalText; }, 1500); downloadVideoFromUrl(baseVideoUrl, fileName); } // --- TOGGLE PLAY/PAUSE VIDEO (UMUM) --- function toggleVideoPlay(videoElement, playBtnElement) { if (!videoElement) return; document.querySelectorAll('video').forEach(v => { if (v !== videoElement && !v.paused) { v.pause(); const otherParent = v.closest('.video-card, .thumb'); if (otherParent) { const otherBtn = otherParent.querySelector('.play-btn'); if (otherBtn) otherBtn.style.display = 'flex'; } } }); if (videoElement.paused) { videoElement.play(); if (playBtnElement) playBtnElement.style.display = 'none'; } else { videoElement.pause(); if (playBtnElement) playBtnElement.style.display = 'flex'; } } // --- SETUP UNTUK VIDEO CARD UTAMA (HORIZONTAL) --- function setupMainVideoCards() { const videoCards = document.querySelectorAll('.video-card'); videoCards.forEach(card => { const video = card.querySelector('video'); const playBtn = card.querySelector('.play-btn'); if (!video || !playBtn) return; video.addEventListener('play', () => { playBtn.style.display = 'none'; }); video.addEventListener('pause', () => { playBtn.style.display = 'flex'; }); video.addEventListener('ended', () => { playBtn.style.display = 'flex'; }); card.addEventListener('click', (e) => { if (e.target.classList && e.target.classList.contains('download-icon')) return; toggleVideoPlay(video, playBtn); }); if (playBtn) { playBtn.addEventListener('click', (e) => { e.stopPropagation(); toggleVideoPlay(video, playBtn); }); } }); } // --- SETUP UNTUK CONTOH (VERTIKAL SCROLL) --- function setupVerticalExample() { const thumbContainers = document.querySelectorAll('.thumb'); thumbContainers.forEach(thumb => { const video = thumb.querySelector('video'); const playBtn = thumb.querySelector('.play-btn'); if (!video || !playBtn) return; video.addEventListener('play', () => { playBtn.style.display = 'none'; }); video.addEventListener('pause', () => { playBtn.style.display = 'flex'; }); video.addEventListener('ended', () => { playBtn.style.display = 'flex'; }); thumb.addEventListener('click', (e) => { if (e.target.classList && e.target.classList.contains('download-icon')) return; document.querySelectorAll('video').forEach(v => { if (v !== video && !v.paused) { v.pause(); const otherParent = v.closest('.video-card, .thumb'); if (otherParent) { const otherBtn = otherParent.querySelector('.play-btn'); if (otherBtn) otherBtn.style.display = 'flex'; } } }); if (video.paused) { video.play(); playBtn.style.display = 'none'; } else { video.pause(); playBtn.style.display = 'flex'; } }); if (playBtn) { playBtn.addEventListener('click', (e) => { e.stopPropagation(); document.querySelectorAll('video').forEach(v => { if (v !== video && !v.paused) { v.pause(); const otherPar = v.closest('.video-card, .thumb'); if (otherPar) { const otherBtn = otherPar.querySelector('.play-btn'); if (otherBtn) otherBtn.style.display = 'flex'; } } }); if (video.paused) { video.play(); playBtn.style.display = 'none'; } else { video.pause(); playBtn.style.display = 'flex'; } }); } }); } // --- SETUP TOMBOL DOWNLOAD KHUSUS UNTUK IKON DI VIDEO --- function setupDownloadButtons() { const downloadBtns = document.querySelectorAll('.download-icon'); downloadBtns.forEach(btn => { btn.addEventListener('click', async (e) => { e.stopPropagation(); let videoSrc = btn.getAttribute('data-video-src'); let fileName = btn.getAttribute('data-video-name'); if (!fileName) fileName = 'video_download.mp4'; if (videoSrc) { if (!fileName.endsWith('.mp4')) fileName += '.mp4'; await downloadVideoFromUrl(videoSrc, fileName); } else { const parentCard = btn.closest('.video-card, .thumb'); if (parentCard) { const videoElem = parentCard.querySelector('video'); if (videoElem && videoElem.src) { let srcUrl = videoElem.src; let name = srcUrl.split('/').pop().split('?')[0]; if (!name.endsWith('.mp4')) name += '.mp4'; await downloadVideoFromUrl(srcUrl, name); } else alert('Video tidak tersedia'); } } }); }); } // Event untuk tombol download dengan resolusi const qualityBtn = document.getElementById('downloadQualityBtn'); if (qualityBtn) { qualityBtn.addEventListener('click', () => { const selectElem = document.getElementById('qualitySelect'); const selectedQuality = selectElem ? selectElem.value : '720'; downloadWithQuality(selectedQuality); }); } // Mencegah drag di video window.addEventListener('dragstart', (e) => { if (e.target.tagName === 'VIDEO' || e.target.closest('.video-card, .thumb')) { e.preventDefault(); } }); document.addEventListener('DOMContentLoaded', () => { setupMainVideoCards(); setupVerticalExample(); setupDownloadButtons(); const allVideos = document.querySelectorAll('video'); allVideos.forEach(vid => { vid.setAttribute('playsinline', 'true'); vid.setAttribute('preload', 'metadata'); }); console.log('✨ Fitur download dengan pilihan resolusi siap | Scroll vertikal untuk contoh video ✨'); }); })();
